\[ \begin{aligned} v_{1} &= c_{1,1} - u_{1} = 4 - 0 = 4 \\ u_{2} &= c_{2,1} - v_{1} = 6 - 4 = 2 \\ v_{3} &= c_{2,3} - u_{2} = 5 - 2 = 3 \\ u_{3} &= c_{3,3} - v_{3} = 5 - 3 = 2 \\ v_{4} &= c_{3,4} - u_{3} = 5 - 2 = 3 \\ u_{4} &= c_{4,4} - v_{4} = 4 - 3 = 1 \\ v_{2} &= c_{4,2} - u_{4} = 2 - 1 = 1 \\ \end{aligned} \]
